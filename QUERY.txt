TABLE CREATION:

CREATE TABLE BUSBOOK_TB (
    RECORDID VARCHAR2(12 BYTE) PRIMARY KEY,
    PASSENGERNAME VARCHAR2(40 BYTE) NOT NULL,
    BUSNUMBER VARCHAR2(20 BYTE) NOT NULL,
    TRAVEL_DATE DATE NOT NULL,
    SEATNO VARCHAR2(10 BYTE),
    TICKETNO VARCHAR2(20 BYTE),
    REMARKS VARCHAR2(100 BYTE)
);


SEQUENCE CREATION:
CREATE SEQUENCE BUSBOOK_SEQ 
START WITH 10 
MAXVALUE 99 
INCREMENT BY 1;
